The Javascript

var config = {
    host: "<server>",
    port: <port (ie. 80, 443)>,
    prefix: "<virtual proxy>",
    isSecure: <whether or not to use SSL>
}
require.config({
    baseUrl: (config.isSecure ? "https://" : "http://") + config.host + (config.port ? ":" + config.port : "") + config.prefix + "resources"
})

require(["js/qlik"], function (qlik) {
    // qlik related js code here
})

Visualization API - Create a Chart

app.visualization.create("barchart|combochart|gauge|kpi|linechart|piechart|pivot-table|scatterplot|table|treemap",columns,options)


app.visualization.create('barchart', [
        "Shape",
        { qDef: { qDef: "=Count([PostedDate])", qLabel: "Sighting Count" } }
    ])
    .then(function (vis) {
        vis.show("vizBasicDiv")
    })

Visualization API - Create an Extension

app.visualization.create(extension name,columns,options)


app.visualization.create('senseui-barchart', [
        "State",
        { qDef: { qDef: "=Count(State)", qLabel: "Sighting Count" } }
    ])
    .then(function (vis) {
        vis.show("vizExtDiv")
    })

Visualization API - Get an Existing Visualization

app.visualization.get(visualization id)


app.visualization.get('CCapKa')
    .then(function (vis) {
        vis.show("vizGetDiv")
    })
